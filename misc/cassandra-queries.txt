CREATE TABLE todo.tasks ( 
   id UUID PRIMARY KEY, 
   name text,
   description text,
   priority text,
   difficulty text,
   start text,
   end text,
   assignee text
    );




CREATE TABLE todo.sub_tasks ( 
   id UUID PRIMARY KEY, 
   task_id text,
   name text,
   description text,
   difficulty text,
   ref text,
   assignee text
    );
    
    
CREATE TABLE todo.users ( 
   id int PRIMARY KEY, 
   name text,
   role text
    );
